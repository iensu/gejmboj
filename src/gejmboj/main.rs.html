1use std::env;
2use std::fs;
3
4use cpu::bus::Bus;
5use cpu::cpu::CPU;
6use cpu::registers::Registers;
7use error::AppResult;
8use log::debug;
9
10mod error;
11
12fn main() -> AppResult<()> {
13    env_logger::init();
14
15    let Some(file_path) = env::args().nth(1) else {
16        panic!("You must provide a path to a Game Boy rom file!");
17    };
18    let rom_bytes = fs::read(&file_path)?;
19
20    debug!("Read {} bytes from {file_path}", rom_bytes.len());
21
22    let mut registers = Registers::new();
23    let mut bus = Bus::new();
24    let mut cpu = CPU::new();
25
26    registers.reset();
27    bus.reset();
28    bus.load(&rom_bytes)?;
29
30    let mut instruction_count = 0;
31
32    loop {
33        instruction_count += 1;
34
35        let prev_pc = registers.PC;
36
37        let instruction = cpu.tick(&mut registers, &mut bus).inspect_err(|e| {
38            eprintln!(
39                "INSTR NO: {instruction_count}, PC: {:04X}, {e}",
40                registers.PC
41            );
42        })?;
43
44        debug!("Executed instruction [{instruction_count:08}] ({prev_pc:04X}) {instruction:?}");
45    }
46}
