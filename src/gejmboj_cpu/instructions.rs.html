35#[derive(Debug, Clone, Copy, PartialEq, Eq)]
36pub enum Condition {
37    Carry,
38    NoCarry,
39    Zero,
40    NotZero,
41}
42
43impl Condition {
44    pub fn parse(c1: u8, c2: u8) -> Result<Self, CpuError> {
45        match (c1, c2) {
46            (0, 0) => Ok(Self::NotZero),
47            (0, 1) => Ok(Self::Zero),
48            (1, 0) => Ok(Self::NoCarry),
49            (1, 1) => Ok(Self::Carry),
50            _ => Err(CpuError::Error(format!(
51                "Unknown instruction condition ({c1}, {c2})"
52            ))),
53        }
54    }
55
56    #[must_use]
57    pub fn is_fulfilled(&self, registers: &Registers) -> bool {
58        match self {
59            Self::Carry => registers.is_carry(),
60            Self::NoCarry => !registers.is_carry(),
61            Self::Zero => registers.is_zero(),
62            Self::NotZero => !registers.is_zero(),
63        }
64    }
65}

75/// Decode an operation code into an `Instruction`.
76#[allow(clippy::too_many_lines)]
77pub fn decode(opcode: u8, pc: u16, memory: &Memory) -> Result<Instruction, CpuError> {
78    match into_bits(opcode) {
79        // ABSOLUTE MATCHES
80        //
81        // misc
82        (0, 0, 0, 0, 0, 0, 0, 0) => Ok(Instruction::Misc(Misc::NOP())),
83        (1, 1, 1, 1, 0, 0, 1, 1) => Ok(Instruction::Misc(Misc::DI())),
84        (1, 1, 1, 1, 1, 0, 1, 1) => Ok(Instruction::Misc(Misc::EI())),
85        (0, 0, 1, 1, 1, 1, 1, 1) => Ok(Instruction::Misc(Misc::CCF())),
86        (0, 0, 1, 1, 0, 1, 1, 1) => Ok(Instruction::Misc(Misc::SCF())),
87        (0, 0, 1, 0, 0, 1, 1, 1) => Ok(Instruction::Misc(Misc::DAA())),
88        (0, 0, 1, 0, 1, 1, 1, 1) => Ok(Instruction::Misc(Misc::CPL())),
89
90        // control flow
91        (1, 1, 0, 0, 0, 0, 1, 1) => Ok(Instruction::ControlFlow(ControlFlow::JP(
92            get_16bit_operand(pc, memory),
93        ))),
94        (1, 1, 0, 0, 1, 0, 0, 1) => Ok(Instruction::ControlFlow(ControlFlow::RET())),
95        (1, 1, 0, 1, 1, 0, 0, 1) => Ok(Instruction::ControlFlow(ControlFlow::RETI())),
96        (1, 1, 0, 0, 1, 1, 0, 1) => Ok(Instruction::ControlFlow(ControlFlow::CALL(
97            get_16bit_operand(pc, memory),
98        ))),
99        (1, 1, 1, 0, 1, 0, 0, 1) => Ok(Instruction::ControlFlow(ControlFlow::JP_HL())),
100        (0, 0, 0, 1, 1, 0, 0, 0) => Ok(Instruction::ControlFlow(ControlFlow::JR(
101            get_8bit_operand(pc, memory),
102        ))),
103
104        // 8 bit load instructions
105        (0, 0, 0, 0, 1, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_BC_TO_A())),
106        (0, 0, 0, 1, 1, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_DE_TO_A())),
107        (0, 0, 0, 0, 0, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_A_TO_BC())),
108        (0, 0, 0, 1, 0, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_A_TO_DE())),
109        (1, 1, 1, 1, 1, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_TO_A(
110            get_16bit_operand(pc, memory),
111        ))),
112        (1, 1, 1, 1, 0, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LDH_C_TO_A())),
113        (1, 1, 1, 0, 0, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LDH_C_FROM_A())),
114        (1, 1, 1, 1, 0, 0, 0, 0) => Ok(Instruction::Load8Bit(Load8Bit::LDH_TO_A(
115            get_8bit_operand(pc, memory),
116        ))),
117        (1, 1, 1, 0, 0, 0, 0, 0) => Ok(Instruction::Load8Bit(Load8Bit::LDH_FROM_A(
118            get_8bit_operand(pc, memory),
119        ))),
120        (1, 1, 1, 0, 1, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_FROM_A(
121            get_16bit_operand(pc, memory),
122        ))),
123        (0, 0, 1, 1, 1, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_A_FROM_HL_DEC())),
124        (0, 0, 1, 1, 0, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_A_TO_HL_DEC())),
125        (0, 0, 1, 0, 1, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_A_FROM_HL_INC())),
126        (0, 0, 1, 0, 0, 0, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_A_TO_HL_INC())),
127        (0, 0, 0, 0, 1, 0, 0, 0) => Ok(Instruction::Load16Bit(Load16Bit::LD_FROM_SP(
128            get_16bit_operand(pc, memory),
129        ))),
130        (1, 1, 1, 1, 1, 0, 0, 1) => Ok(Instruction::Load16Bit(Load16Bit::LD_HL_TO_SP())),
131
132        // ALU 8-bit instructions
133        (1, 0, 0, 0, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::ADD_HL())),
134        (1, 1, 0, 0, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::ADD_N(get_8bit_operand(
135            pc, memory,
136        )))),
137        (1, 0, 0, 0, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::ADC_HL())),
138        (1, 1, 0, 0, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::ADC_N(get_8bit_operand(
139            pc, memory,
140        )))),
141        (1, 0, 0, 1, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::SUB_HL())),
142        (1, 1, 0, 1, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::SUB_N(get_8bit_operand(
143            pc, memory,
144        )))),
145        (1, 0, 0, 1, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::SBC_HL())),
146        (1, 1, 0, 1, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::SBC_N(get_8bit_operand(
147            pc, memory,
148        )))),
149        (1, 0, 1, 0, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::AND_HL())),
150        (1, 1, 1, 0, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::AND_N(get_8bit_operand(
151            pc, memory,
152        )))),
153        (1, 0, 1, 1, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::OR_HL())),
154        (1, 1, 1, 1, 0, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::OR_N(get_8bit_operand(
155            pc, memory,
156        )))),
157        (1, 0, 1, 0, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::XOR_HL())),
158        (1, 1, 1, 0, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::XOR_N(get_8bit_operand(
159            pc, memory,
160        )))),
161        (1, 0, 1, 1, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::CP_HL())),
162        (1, 1, 1, 1, 1, 1, 1, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::CP_N(get_8bit_operand(
163            pc, memory,
164        )))),
165        (0, 0, 1, 1, 0, 1, 0, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::INC_HL())),
166        (0, 0, 1, 1, 0, 1, 0, 1) => Ok(Instruction::ALU8Bit(ALU8Bit::DEC_HL())),
167
168        // ALU 16-bit instructions
169        (1, 1, 1, 0, 1, 0, 0, 0) => Ok(Instruction::ALU16Bit(ALU16Bit::ADD_SP(get_8bit_operand(
170            pc, memory,
171        )))),
172
173        // Rotate Shift instructions
174        (0, 0, 0, 0, 0, 1, 1, 1) => Ok(Instruction::RotateShift(RotateShift::RLCA())),
175        (0, 0, 0, 0, 1, 1, 1, 1) => Ok(Instruction::RotateShift(RotateShift::RRCA())),
176        (0, 0, 0, 1, 0, 1, 1, 1) => Ok(Instruction::RotateShift(RotateShift::RLA())),
177        (0, 0, 0, 1, 1, 1, 1, 1) => Ok(Instruction::RotateShift(RotateShift::RRA())),
178        (1, 1, 0, 0, 1, 0, 1, 1) => {
179            let operand = get_8bit_operand(pc, memory);
180
181            rotate_shift::decode(operand)
182                .map(Instruction::RotateShift)
183                .or_else(|_| bit::decode(operand).map(Instruction::Bit))
184        }
185
186        (1, 1, 1, 1, 1, 0, 0, 0) => Ok(Instruction::Stack(Stack::LD_HL_SP_E(get_8bit_operand(
187            pc, memory,
188        )))),
189
190        // VARIABLE MATCHES
191        //
192        // control flow
193        (1, 1, 0, c, d, 0, 1, 0) => Ok(Instruction::ControlFlow(ControlFlow::JPC(
194            get_16bit_operand(pc, memory),
195            Condition::parse(c, d).unwrap(),
196        ))),
197        (0, 0, 1, c, d, 0, 0, 0) => Ok(Instruction::ControlFlow(ControlFlow::JRC(
198            get_8bit_operand(pc, memory),
199            Condition::parse(c, d).unwrap(),
200        ))),
201        (1, 1, 0, c, d, 1, 0, 0) => Ok(Instruction::ControlFlow(ControlFlow::CALLC(
202            get_16bit_operand(pc, memory),
203            Condition::parse(c, d).unwrap(),
204        ))),
205        (1, 1, 0, c, d, 0, 0, 0) => Ok(Instruction::ControlFlow(ControlFlow::RETC(
206            Condition::parse(c, d).unwrap(),
207        ))),
208        (1, 1, _, _, _, 1, 1, 1) => Ok(Instruction::ControlFlow(ControlFlow::RST(opcode))),
209
210        // 8 bit load instructions
211        (0, 1, a, b, c, 1, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_FROM_HL(
212            (a, b, c).into(),
213        ))),
214        (0, 1, 1, 1, 0, a, b, c) => Ok(Instruction::Load8Bit(Load8Bit::LD_TO_HL((a, b, c).into()))),
215        (0, 1, x1, y1, z1, x2, y2, z2) => Ok(Instruction::Load8Bit(Load8Bit::LD(
216            (x1, y1, z1).into(),
217            (x2, y2, z2).into(),
218        ))),
219        (0, 0, a, b, c, 1, 1, 0) => Ok(Instruction::Load8Bit(Load8Bit::LD_IMM(
220            (a, b, c).into(),
221            get_8bit_operand(pc, memory),
222        ))),
223
224        // 16 bit load instructions
225        (0, 0, a, b, 0, 0, 0, 1) => Ok(Instruction::Load16Bit(Load16Bit::LD(
226            (0, a, b).into(),
227            get_16bit_operand(pc, memory),
228        ))),
229        (1, 1, a, b, 0, 1, 0, 1) => Ok(Instruction::Load16Bit(Load16Bit::PUSH((1, a, b).into()))),
230        (1, 1, a, b, 0, 0, 0, 1) => Ok(Instruction::Load16Bit(Load16Bit::POP((1, a, b).into()))),
231
232        // ALU 8-bit instructions
233        (1, 0, 0, 0, 0, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::ADD((a, b, c).into()))),
234        (1, 0, 0, 0, 1, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::ADC((a, b, c).into()))),
235        (1, 0, 0, 1, 0, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::SUB((a, b, c).into()))),
236        (1, 0, 0, 1, 1, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::SBC((a, b, c).into()))),
237        (1, 0, 1, 0, 0, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::AND((a, b, c).into()))),
238        (1, 0, 1, 1, 0, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::OR((a, b, c).into()))),
239        (1, 0, 1, 0, 1, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::XOR((a, b, c).into()))),
240        (1, 0, 1, 1, 1, a, b, c) => Ok(Instruction::ALU8Bit(ALU8Bit::CP((a, b, c).into()))),
241        (0, 0, a, b, c, 1, 0, 0) => Ok(Instruction::ALU8Bit(ALU8Bit::INC((a, b, c).into()))),
242        (0, 0, a, b, c, 1, 0, 1) => Ok(Instruction::ALU8Bit(ALU8Bit::DEC((a, b, c).into()))),
243
244        // ALU 16-bit instructions
245        (0, 0, b, c, 1, 0, 0, 1) => Ok(Instruction::ALU16Bit(ALU16Bit::ADD_HL((0, b, c).into()))),
246        (0, 0, b, c, 0, 0, 1, 1) => Ok(Instruction::ALU16Bit(ALU16Bit::INC((0, b, c).into()))),
247        (0, 0, b, c, 1, 0, 1, 1) => Ok(Instruction::ALU16Bit(ALU16Bit::DEC((0, b, c).into()))),
248
249        // Catch all
250        _ => Err(CpuError::UnknownInstruction(opcode)),
251    }
